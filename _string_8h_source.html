38 #if defined(OS_WIN32) && !defined(PLATFORM_MINGW)
39 // MSVC does not have snprintf, and has a vsnprintf that does not have the same
40 // semantics as the linux one, which means we have to provide wrappers for
41 // both.
42 static inline int VSNPrintf(char *str, size_t size, char const *format,
43  va_list ap) {
44  int res = -1;
45  if (size > 0) {
46  res = _vsnprintf_s(str, size, _TRUNCATE, format, ap);
47  }
48  if (res == -1) {
49  res = _vscprintf(format, ap);
50  }
51  return res;
52 }
53 #elif defined(OS_POSIX) || (defined(OS_WIN32) && defined(PLATFORM_MINGW))
54 // The posix systems we support, as well as MinGW (which provides a gnu
55 // environment on Windows, with GNU semantics) have a sane vsnprintf. Use that.
56 static inline int VSNPrintf(char *str, size_t size, char const *format,
57  va_list ap) {
58  return ::vsnprintf(str, size, format, ap);
59 }
60 #endif

62 static inline int SNPrintf(char *str, size_t size, char const *format, ...) {
63  va_list ap;
64  va_start(ap, format);
65  int res = VSNPrintf(str, size, format, ap);
66  va_end(ap);
67  return res;
